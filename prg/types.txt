// Variadic args
extern fn i32 printf: s char&, ... ;

// Floats
let a = 0.1 + 0.2;
let b = a + 4.0;

// Pointers
let deref = 5;
let ptr = #deref;

// Strings
let c = "Hello, World!";

fn f64 float: d f64, e char& {
    return d + 1.0;
}

// Returning pointers
fn char& str {
    return "Hello, World! %f";
}

// Nested function calls
printf(str(), float(b, c));

// Checking the deref
printf("%d", @ptr);

return 0;
